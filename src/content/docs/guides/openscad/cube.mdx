<CardGrid>

<Card title="Simple Uniform Cube">
<Tabs>
  <TabItem label="Usage">
    ```js
    cube(size = x);
    ```
  </TabItem>
  <TabItem label="Example">
    ```js
    cube(10);
    ```
    ![A simple 10mm uniform cube rendered in OpenSCAD](https://cdn.todoprogramming.org/openscad/10mm-cube.png)
  </TabItem>
</Tabs>
</Card>

<Card title="Non Uniform Cube">
<Tabs>
  <TabItem label="Usage">
    ```js
    cube(size = [x,y,z]);
    ```
  </TabItem>
  <TabItem label="Example">
    ```js
    cube([25,35,55]);
    ```
    ![A non-uniform cube with dimensions 25x35x55mm rendered in OpenSCAD](https://cdn.todoprogramming.org/openscad/non-uniform-cube.png)
  </TabItem>
</Tabs>
</Card>

<Card title="Centered Cube">
<Tabs>
  <TabItem label="Usage">
    ```js
    cube(x, center=true)
    cube([x,y,z],center=true);
    ```
  </TabItem>
  <TabItem label="Example">
    ```js
    cube([20,30,50],center=true);
    ```
    ![A centered cube with dimensions 20x30x50mm rendered in OpenSCAD](https://cdn.todoprogramming.org/openscad/centered-cube.png)
  </TabItem>
</Tabs>
</Card>
